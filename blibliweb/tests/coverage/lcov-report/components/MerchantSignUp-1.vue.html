<script>

export default {
  data() {
    return {
      email: '',
      emailErrorMsg: '',
      state: null,
      password: '',
      password2: '',
      passwordState: null,
      passwordErrorMsg: '',
      samePasswordState: null,
      samePasswordErrorMsg: '',
      ktp: '',
      nama: '',
      notlp: '',
      tgl: '',
      photo: [],
      btnState: false,
    };
  },
  computed: {
    emailState() {
      return this.email.length > 0;
    },
  },
  methods: {
    emailCheckFormat() {
      /* eslint-disable */
      const re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
      /* eslint-enable */
      this.state = re.test(this.email);
      this.emailErrorMsg = 'Email anda salah.';
      this.checkAllInput();
    },
    passwordCheckFormat() {
      const re = new RegExp('^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.{8,})');
      this.passwordState = re.test(this.password);
      if (!re.test(this.password)) this.passwordErrorMsg = 'Format password tidak sesuai';
      else this.passwordErrorMsg = '';
      this.samePasswordCheck();
    },
    samePasswordCheck() {
      if (this.password !== this.password2) {
        this.samePasswordState = false;
        this.samePasswordErrorMsg = 'Password tidak sama';
      } else {
        this.samePasswordErrorMsg = '';
        this.samePasswordState = true;
      }
      this.checkAllInput();
    },
    checkAllInput() {
      if (
        this.email !== ''
        && this.password !== ''
        && this.password2 !== ''
        && this.nama !== ''
        && this.ktp !== ''
        && this.notlp !== ''
        && this.tgl !== ''
      ) {
        this.btnState = true;
        this.$refs.btn.disabled = false;
      } else {
        this.btnState = false;
        this.$refs.btn.disabled = true;
      }
    },
    next() {
      this.tahap1 = true;
      this.$router.push('langkah-2');
      console.log('sad');
      console.log(this.tahap1);
    },
    processFile(event) {
      /* eslint-disable */
      this.photo = event.target.files[0];
    },
  },
